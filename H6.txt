function checkProbabilityTheory(count) {
    let evenCount = 0;
    let oddCount = 0;

    for (let i = 0; i < count; i++) {
        let randomNum = Math.floor(Math.random() * (1000 - 100 + 1)) + 100;  
        if (randomNum % 2 === 0) {
            evenCount += 1;
        } else {
            oddCount += 1;
        }
    }

    let totalCount = evenCount + oddCount;
    let evenPercentage = (evenCount / totalCount) * 100;
    let oddPercentage = (oddCount / totalCount) * 100;

    console.log(`Кількість згенерованих чисел: ${totalCount}`);
    console.log(`Кількість парних чисел: ${evenCount}`);
    console.log(`Кількість непарних чисел: ${oddCount}`);
    console.log(`% парних до непарних: ${evenPercentage.toFixed(2)}% : ${oddPercentage.toFixed(2)}%`);
}

checkProbabilityTheory(1000);
