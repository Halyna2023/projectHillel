// Створюємо об'єкт "services"
var services = {
    "стрижка": "60 грн",
    "гоління": "80 грн",
    "Миття голови": "100 грн",
    // Метод для обчислення загальної вартості наданих послуг
    price: function() {
        let total = 0;
        for (let key in this) {
            if (this.hasOwnProperty(key) && typeof this[key] === 'string' && this[key].endsWith('грн')) {
                total += parseInt(this[key]);
            }
        }
        return total + ' грн';
    },
    // Метод для знаходження мінімальної вартості послуг
    minPrice: function() {
        let min = Infinity;
        for (let key in this) {
            if (this.hasOwnProperty(key) && typeof this[key] === 'string' && this[key].endsWith('грн')) {
                let price = parseInt(this[key]);
                if (price < min) {
                    min = price;
                }
            }
        }
        return min + ' грн';
    },
    // Метод для знаходження максимальної вартості послуг
    maxPrice: function() {
        let max = -Infinity;
        for (let key in this) {
            if (this.hasOwnProperty(key) && typeof this[key] === 'string' && this[key].endsWith('грн')) {
                let price = parseInt(this[key]);
                if (price > max) {
                    max = price;
                }
            }
        }
        return max + ' грн';
    }
};

// Додаємо нову послугу
services['Розбити скло'] = "200 грн";

// Викликаємо методи для перевірки
console.log('Загальна вартість: ' + services.price()); // Загальна вартість: 440 грн
console.log('Мінімальна вартість: ' + services.minPrice()); // Мінімальна вартість: 60 грн
console.log('Максимальна вартість: ' + services.maxPrice()); // Максимальна вартість: 200 грн
